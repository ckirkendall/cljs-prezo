Example 1: JQuery Plugin (JavaScript)

The Logger Function 
(function($){
   $.fn.extend({
       //plugin name - logger
        logger: function(options) {

           var defaults = {
               pre : "Log:"
           };

           var options = $.extend(defaults, options);

           return this.each(function() {
                 var obj = $(this);
                 console.log(pre, obj);
           });
       }
   });
})(jQuery);
Calling Our Logger 
JQuery.select(".cool")
      .logger({ pre : "logger" })
      .append("span")
      .text("hello")
